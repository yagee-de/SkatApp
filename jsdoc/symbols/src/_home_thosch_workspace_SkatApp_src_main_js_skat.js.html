388             gamePoints : function(game) {
389               var gamePoints, gameType = game.gameType;
390               if (game.bid === 0) {
391                 if (game.gameLevel === 2) {
392                   // Durchmarsch
393                   return 240;
394                 }
395                 return game.points * game.gameLevel;
396               } else if (gameType === 23 || gameType === 35 || gameType === 46 || gameType === 59) {
397                 gamePoints = gameType;
398               } else {
399                 gamePoints = gameType * (game.jacks + game.gameLevel) * game.announcement;
400               }
401               if (!game.won) {
402                 gamePoints *= -2;
403               }
404               return gamePoints;
405             },

458             apiCall : function(path, success) {

460               if (dbURL === null || dbURL.length === 0) {
461                 window.alert("Keine SKatDB-URL definiert");
462                 return;
463               }
464               url = dbURL + path;
465               jQuery.ajax({
466                 url : url,
467                 type : 'GET',
468                 crossDomain : true,
469                 dataType : 'json',
470                 error : function(xhr, status) {
471                   window.alert("Fehler: " + xhr.status + " (" + xhr.statusText + ")");
472                   if (window.confirm('URL "' + url + '" laden?')) {

498                 },
499                 dataType : 'json',

512                   }

514                 }
515               });
516             }
