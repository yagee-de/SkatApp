 18 define("Skat",
 19     [ "jquery", "jqm-init", "jquery.mobile", "Class", "SkatForm", "SkatGame", "SkatGames", "SkatSettings", "SkatSync" ],
 20     function(jQuery, jqmInit, jqm, Class, SkatForm, SkatGame, SkatGames, SkatSettings, SkatSync) {
 21       "use strict";
 22       var Skat = Class
 23           .extend(
 24           /** @lends Skat.prototype */
 25           {
 26             /** 
 27              * @constructs
 28              * @extends Class 
 29              */
 30             init : function() {
 31               this.form = new SkatForm();
 32               this.games = new SkatGames();
 33               this.settings = new SkatSettings();
 34               this.sync = new SkatSync();
 35             },
 36             /**
 37              * @memberOf Skat#
 38              * @description logic for form page
 39              * @type SkatForm
 40              */
 41             form : null,
 42             /**
 43              * @memberOf Skat#
 44              * @description logic for games page
 45              * @type SkatGames
 46              */
 47             games : null,
 48             /**
 49              * @memberOf Skat#
 50              * @description logic for settings page
 51              * @type SkatSettings
 52              */
 53             settings : null,
 54             /**
 55              * @memberOf Skat#
 56              * @description logic for sync page
 57              * @type SkatSync
 58              */
 59             sync : null,
 60             /**
 61              * @description handles events when clicked on links with 'data-preparegame' attributes
 62              * @see SkatForm#prepareGame
 63              * @param {Event} e the click event
 64              */
 65             prepareGameHandler : function(e) {
 66               var game = parseInt(jQuery(e.currentTarget).attr("data-preparegame"), 10);
 67               this.form.prepareGame(game);
 68             }
 69           });
 70       return Skat;
 71     });
