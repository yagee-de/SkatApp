 18 define("SkatStorage", [ "jquery", "jqm-init", "jquery.mobile", "Class", "SkatGame" ], function(jQuery, jqmInit, jqm, Class, SkatGame) {
 19   "use strict";
 20   var SkatStorage = Class.extend(
 21   /** @lends SkatStorage.prototype */
 22   {
 23     /** 
 24      * @constructs
 25      * @extends Class 
 26      */
 27     init : function() {
 28     },
 29     /**
 30      * @memberOf SkatStorage#
 31      * @constant
 32      * @description storage keys for localStorage
 33      *              <dl>
 34      *              <dt>games:</dt>
 35      *              <dd>uncommitted games</dd>
 36      *              <dt>players:</dt>
 37      *              <dd>all player names</dd>
 38      *              <dt>groups:</dt>
 39      *              <dd>all group names</dd>
 40      *              <dt>dbURL:</dt>
 41      *              <dd>URL do SkatDB server</dd>
 42      *              </dl>
 43      */
 44     storage : {
 45       games : "games",
 46       players : "players",
 47       groups : "groups",
 48       dbURL : "dbURL"
 49     },
 50     /**
 51      * @description load stored object from local storage
 52      * @param {String} storeKey key of object in local storage
 53      * @returns {Object}
 54      */
 55     load : function(storeKey) {
 56       return JSON.parse(window.localStorage.getItem(this.storage[storeKey]));
 57     },
 58     /**
 59      * @description save object to local storage
 60      * @param {String} storeKey key of object in local storage
 61      * @param {Object} value object to store
 62      */
 63     store : function(storeKey, value) {
 64       window.localStorage.setItem(this.storage[storeKey], JSON.stringify(value));
 65     },
 66     /**
 67      * @description deletes game with given number after user confirm
 68      * @param {int} gameNumber number of game (counting from 0)
 69      */
 70     deleteGame : function(gameNumber) {
 71       var games = this.load("games") || [];
 72       if (gameNumber < 0 || gameNumber >= games.length) {
 73         return false;
 74       }
 75       if (window.confirm('Spiel Nr.' + (gameNumber + 1) + ' wirklich l√∂schen?')) {
 76         games.splice(gameNumber, 1);
 77         this.store("games", games);
 78         return true;
 79       }
 80       return false;
 81     },
 82     /**
 83      * @description delete all stored games
 84      */
 85     removeGames : function() {
 86       this._skat.store("games", []);
 87       return true;
 88     },
 89     /**
 90      * @description clear local storage (removing dbURL, games, players and groups)
 91      */
 92     clearStorage : function() {
 93       window.localStorage.clear();
 94     }
 95   });
 96   return SkatStorage;
 97 });
