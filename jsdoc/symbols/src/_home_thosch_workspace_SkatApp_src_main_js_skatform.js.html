 18 define("SkatForm",
 19     [ "jquery", "jqm-init", "jquery.mobile", "SkatGame", "SkatStorage" ],
 20     function(jQuery, jqmInit, jqm, SkatGame, SkatStorage) {
 21       "use strict";
 22       var SkatForm = SkatStorage
 23           .extend(
 24           /** @lends SkatForm.prototype */
 25           {
 26             /** 
 27              * @constructs
 28              * @extends SkatStorage 
 29              */
 30             init : function() {
 31             },
 32             /**
 33              * @memberOf SkatForm#
 34              * @type int
 35              * @description current game from localStore or '-1' for new game
 36              */
 37             currentGame : -1,
 38             /**
 39              * @field
 40              * @type boolean
 41              * @description if 'false' prevents {@link SkatForm#updateGameLevel} to refresh jquery mobile widgets
 42              */
 43             createComplete : false,
 44             /**
 45              * @description initialize bid form element with every valid value (call this once on #formPage create)
 46              */
 47             initBiddingValues : function() {
 48               var biddingSelector = '#bid';
 49               jQuery(biddingSelector).empty();
 50               jQuery(biddingSelector).append('<option value="0">Ramsch</option>');
 51               jQuery.each(SkatGame.bids, function(i, value) {
 52                 var option = jQuery('<option>').text(value).attr('value', value);
 53                 if (i === 0) {
 54                   option.attr('selected', true);
 55                 }
 56                 jQuery(biddingSelector).append(option);
 57               });
 58             },
 59             /**
 60              * @description initialize players form element with every valid value (call this before displaying #formPage)
 61              */
 62             initPlayers : function() {
 63               var players = (this.load("players") || []), playersSelector = "#player", option;
 64               jQuery(playersSelector).empty();
 65               option = jQuery('<option>').attr('selected', true);
 66               jQuery(playersSelector).append(option);
 67               jQuery.each(players, function(i, player) {
 68                 option = jQuery('<option>').text(player).attr('value', player);
 69                 jQuery(playersSelector).append(option);
 70               });
 71             },
 72             /**
 73              * @description initialize groups form element with every valid value (call this before displaying #formPage)
 74              */
 75             initGroups : function() {
 76               var groups = (this.load("groups") || []), groupsSelector = "#group";
 77               jQuery(groupsSelector).empty();
 78               jQuery.each(groups, function(i, group) {
 79                 var option = jQuery('<option>').text(group).attr('value', group);
 80                 if (i === 0) {
 81                   option.attr('selected', true);
 82                 }
 83                 jQuery(groupsSelector).append(option);
 84               });
 85             },
 86             /**
 87              * @description initialize jacks form element with every valid value (call this once on #formPage create)
 88              */
 89             initJacks : function() {
 90               var jacksSelector = "#jacks", i, option;
 91               jQuery(jacksSelector).empty();
 92               for (i = 1; i < 12; i++) {
 93                 option = jQuery('<option>').text(i).attr('value', i);
 94                 if (i === 0) {
 95                   option.attr('selected', true);
 96                 }
 97                 jQuery(jacksSelector).append(option);
 98               }
 99             },
100             /**
101              * @description acts as a setter for {@link SkatForm#currentGame}
102              */
103             prepareGame : function(gameNumber) {
104               this.currentGame = typeof gameNumber === 'number' ? gameNumber : -1;
105             },
106             /**
107              * @description uses data of specified stored {@link SkatGame} to fill form elements (run refreshForm after this)
108              * @see SkatForm#refreshForm  
109              */
110             fillForm : function(gameNumber) {
111               if (gameNumber < 0) {
112                 jQuery("#deleteButton").parentsUntil("#inputForm", "div.ui-btn").hide();
113                 jQuery("#formPage").children(":jqmData(role=header)").children("h1").text("Spiel anlegen");
114                 return;
115               }
116               var games = this.load("games") || [], game;
117               if (gameNumber >= games.length) {
118                 return;
119               }
120               game = new SkatGame(games[gameNumber]);
121               jQuery("#formPage").children(":jqmData(role=header)").children("h1").text("Spiel Nr. " + (gameNumber + 1) + " bearbeiten");
122               jQuery("#won").attr("checked", game.won ? true : false);
123               jQuery("#hand").attr("checked", game.hand ? true : false);
124               this.updateGameLevel();
125               jQuery("#group").val(game.group);
126               jQuery("#player").val(game.player);
127               jQuery("#bid").val(game.bid).trigger('onchange');
128               jQuery("#jacks").val(game.jacks);
129               jQuery("#gameType").val(game.gameType);
130               jQuery("#gameLevel").val(game.gameLevel);
131               jQuery("#announcement").val(game.announcement);
132               jQuery("#timeCreated").val(game.createDate.toISOString());
133               jQuery("#points").val(game.points ? game.points : 40);
134               jQuery("#deleteButton").parentsUntil("#inputForm", "div.ui-btn").show();
135             },
136             /**
137              * @description resets this form element with reasonable defaults
138              */
139             resetForm : function() {
140               jQuery("#won").attr("checked", true);
141               jQuery("#hand").attr("checked", false);
142               this.updateGameLevel();
143               jQuery("#player").val(jQuery(this).prop("defaultSelected"));
144               jQuery("#bid").val(18).trigger('onchange');
145               jQuery("#jacks").val(1);
146               jQuery("#gameType").val(9);
147               jQuery("#gameLevel").val(1);
148               jQuery("#announcement").val(1);
149               if (jQuery("#timeCreated").val().length === 0) {
150                 jQuery("#timeCreated").val((new Date()).toISOString());
151               }
152               jQuery("#points").val(40);
153             },
154             /**
155              * @description refreshes jquery mobile widgets after form element values changed
156              */
157             refreshForm : function() {
158               jQuery("#won").checkboxradio("refresh");
159               jQuery("#hand").checkboxradio("refresh");
160               jQuery("#player").selectmenu("refresh");
161               jQuery("#group").selectmenu("refresh");
162               jQuery("#bid").selectmenu("refresh");
163               jQuery("#jacks").selectmenu("refresh");
164               jQuery("#gameType").selectmenu("refresh");
165               jQuery("#gameLevel").selectmenu("refresh");
166               jQuery("#announcement").selectmenu("refresh");
167               jQuery("#points").slider("refresh");
168             },
169             /**
170              * @description calculates score on current form element values and updates page
171              */
172             updateScore : function() {
173               var game = this.getGame(), gamePoints = game.getPoints();
174               document.getElementById("p1").innerHTML = "Punkte: " + gamePoints;
175             },
176             /**
177              * @description switches between Ramsch mode and normal game mode
178              * @param {HTMLElement} source bid select form element
179              */
180             bidChange : function(source) {
181               var bid = parseInt(source.value, 10);
182               switch (bid) {
183               case 0:
184                 jQuery("#points").parentsUntil("#inputForm", "div.ui-field-contain").show();
185                 jQuery("#hand").parentsUntil("#inputForm", "div.ui-field-contain").hide();
186                 jQuery("#gameType").parentsUntil("#inputForm", "div.ui-field-contain").hide();
187                 jQuery("#announcement").parentsUntil("#inputForm", "div.ui-field-contain").hide();
188                 jQuery("#jacks").parentsUntil("#inputForm", "div.ui-field-contain").hide();
189                 jQuery("#won").parentsUntil("#inputForm", "div.ui-field-contain").hide();
190                 break;
191 
192               default:
193                 jQuery("#points").parentsUntil("#inputForm", "div.ui-field-contain").hide();
194                 jQuery("#hand").parentsUntil("#inputForm", "div.ui-field-contain").show();
195                 jQuery("#gameType").parentsUntil("#inputForm", "div.ui-field-contain").show();
196                 jQuery("#announcement").parentsUntil("#inputForm", "div.ui-field-contain").show();
197                 jQuery("#jacks").parentsUntil("#inputForm", "div.ui-field-contain").show();
198                 jQuery("#won").parentsUntil("#inputForm", "div.ui-field-contain").show();
199                 break;
200               }
201               this.updateGameLevel();
202             },
203             /**
204              * @description updates game level element which depends on bid or hand play
205              */
206             updateGameLevel : function() {
207               if (parseInt(jQuery("#bid").val(), 10) === 0) {
208                 document.getElementById("gameLevel").innerHTML = "<option value=-1>Normal</option><option value=-2>Jungfer (*-2)</option><option value=2>Durchmarsch (*2)</option>";
209               } else if (jQuery("#hand").attr('checked')) {
210                 document.getElementById("gameLevel").innerHTML = "<option value=2>Hand (+2)</option><option value=3>Schneider (+3)</option><option value=4>Schneider angesagt (+4)</option><option value=5>Schwarz (+5)</option><option value=6>Schwarz angesagt (+6)</option><option value=7>Ouvert (+7)</option>";
211               } else {
212                 document.getElementById("gameLevel").innerHTML = "<option value=1>Normal (+1)</option><option value=2>Schneider (+2)</option><option value=3>Schwarz (+3)</option>";
213               }
214               if (this.createComplete) {
215                 jQuery("#gameLevel").selectmenu("refresh");
216               }
217             },
218             /**
219              * @description builds a {@link SkatGame} instance on current form element values
220              * @returns {SkatGame}
221              */
222             getGame : function() {
223               var group = jQuery("#group").val(), player = jQuery("#player").val(), bid = parseInt(jQuery("#bid").val(), 10), gamelevel = parseInt(jQuery("#gameLevel")
224                   .val(),
225                   10), modified = new Date().getTime(), created = modified, timeCreated, json, points;
226               try {
227                 timeCreated = jQuery("#timeCreated").val();
228                 if (timeCreated.length > 0) {
229                   timeCreated = new Date(timeCreated);
230                   created = timeCreated.getTime();
231                 }
232               } catch (e) {
233                 window.alert(e);
234               }
235               json = {
236                 "group" : group,
237                 "player" : player,
238                 "bid" : bid,
239                 "gameLevel" : gamelevel,
240                 "createDate" : created,
241                 "modifyDate" : modified
242               };
243               if (bid === 0) {
244                 // Ramsch
245                 points = parseInt(jQuery("#points").val(), 10);
246                 json.points = (gamelevel === 2) ? 120 : points;
247                 json.won = (gamelevel === 2);
248               } else {
249                 json.jacks = parseInt(jQuery("#jacks").val(), 10);
250                 json.gameType = parseInt(jQuery("#gameType").val(), 10);
251                 json.hand = jQuery("#hand").attr('checked') ? true : false;
252                 json.announcement = parseInt(jQuery("#announcement").val(), 10);
253                 json.won = jQuery("#won").attr('checked') ? true : false;
254 
255               }
256               return new SkatGame(json);
257             },
258             storeGame : function() {
259               var game = this.getGame().toJSON(), games;
260               if (game.player.length === 0) {
261                 window.alert("Kein Spieler ausgewählt");
262                 jQuery.mobile.silentScroll(0);
263                 return;
264               }
265               games = this.load("games") || [];
266               if (this.currentGame === -1) {
267                 games.push(game);
268               } else {
269                 games[this.currentGame] = game;
270               }
271               this.store("games", games);
272               window.alert("Spiel gespeichert");
273               this.resetForm();
274               this.refreshForm();
275               if (this.currentGame === -1) {
276                 jQuery.mobile.silentScroll(0);
277               } else {
278                 window.history.back();
279               }
280             }
281           });
282       return SkatForm;
283     });
