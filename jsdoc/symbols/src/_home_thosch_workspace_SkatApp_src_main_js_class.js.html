1 /*global define,window */

  6 // Inspired by base2 and Prototype
  7 // modified by Thomas Scheffler to work with jslint
  8 define("Class", [ "jquery" ], function(jQuery) {
  9 	"use strict";
 10 	var initializing = false, Class;
 11 
 12 	// The base Class implementation (does nothing)
 13 	/** @class allows to make object oriented class hierarchies */
 14 	Class = function() {
 15 	};
 16 
 17 	// Create a new Class that inherits from this class
 18 	/**  @memberOf Class
 19 	 *   @param {Object} prop Object with properties attached to prototype
 20 	 *   @see <a href="http://ejohn.org/blog/simple-javascript-inheritance/">authors site</a> */
 21 	Class.extend = function extend(prop) {
 22 		var _super = this.prototype, Constructor = this, prototype, name;
 23 
 24 		// Instantiate a base class (but only create the instance,
 25 		// don't run the init constructor)
 26 		initializing = true;
 27 		prototype = new Constructor();
 28 		initializing = false;
 29 
 30 		// Copy the properties over onto the new prototype
 31 		jQuery.each(prop, function(name, value) {
 32 			// Check if we're overwriting an existing function
 33 			prototype[name] = typeof value === "function"
 34 					&& typeof _super[name] === "function" ? function() {
 35 				var tmp = this._super, ret;
 36 
 37 				// Add a new ._super() method that is the same
 38 				// method
 39 				// but on the super-class
 40 				this._super = _super[name];
 41 
 42 				// The method only need to be bound temporarily, so
 43 				// we
 44 				// remove it when we're done executing
 45 				ret = value.apply(this, arguments);
 46 				this._super = tmp;
 47 
 48 				return ret;
 49 			} : value;
 50 		});
 51 
 52 		/** @description The dummy class constructor*/
 53 		function Class() {
 54 			// All construction is actually done in the init method
 55 			if (!initializing && this.init) {
 56 				this.init.apply(this, arguments);
 57 			}
 58 		}
 59 
 60 		// Populate our constructed prototype object
 61 		Class.prototype = prototype;
 62 
 63 		// Enforce the constructor to be what we expect
 64 		Class.prototype.constructor = Class;
 65 
 66 		// And make this class extendable
 67 		Class.extend = extend;
 68 
 69 		return Class;
 70 	};
 71 	return Class;
 72 });
