 18 define("SkatSync", [ "jquery", "jqm-init", "jquery.mobile", "SkatStorage", "SkatGame" ], function(jQuery, jqmInit, jqm, SkatStorage,
 19                                                                                                   SkatGame) {
 20   "use strict";
 21   var api, SkatSync;
 22   /**
 23    * @memberOf SkatSync
 24    * @constant
 25    * @description URIs relative to storage.dbURL for syncing with SkatDB server
 26    *              <dl>
 27    *              <dt>players:</dt>
 28    *              <dd>get player from SkatDB</dd>
 29    *              <dt>groups:</dt>
 30    *              <dd>get groups from SkatDB</dd>
 31    *              <dt>games:</dt>
 32    *              <dd>post games to SkatDB</dd>
 33    *              </dl>
 34    */
 35   api = {
 36     groups : "JSON/groups",
 37     players : "JSON/players",
 38     games : "JSON/games"
 39   };
 40   SkatSync = SkatStorage.extend(
 41   /** @lends SkatSync.prototype */
 42   {
 43     /**
 44      * @constructs
 45      * @extends SkatStorage
 46      */
 47     init : function() {
 48     },
 49     /**
 50      * @description generated exportable JSON object of all stored games
 51      * @returns {Array} of games
 52      */
 53     getExport : function() {
 54       var games = this.load("games") || [];
 55       jQuery.each(games, function(i, game) {
 56         var value = new SkatGame(game).getPoints();
 57         game.value = value;
 58       });
 59       return games;
 60     },
 61     /**
 62      * @description downloads players from SkatDB and overwrites list in local storage
 63      */
 64     downloadPlayer : function() {
 65       this.apiCall(api.players, function(players) {
 66         players.sort();
 67         this.store("players", players);
 68         window.alert(players.length + " Spieler geladen.");
 69       });
 70     },
 71     /**
 72      * @description downloads groups from SkatDB and overwrites list in local storage
 73      */
 74     downloadGroups : function() {
 75       this.apiCall(api.groups, function(groups) {
 76         groups.sort();
 77         this.store("groups", groups);
 78         window.alert(groups.length + " Gruppen geladen.");
 79       });
 80     },
 81     /**
 82      * @description helper methods for GET requests to SkatDB
 83      * @param {String}
 84      *          path relative path to SkatDB url
 85      * @param {function}
 86      *          success function that should be called on success with requested json as parameter
 87      */
 88     apiCall : function(path, success) {
 89       var dbURL = this.load("dbURL"), url, that = this;
 90       if (dbURL === null || dbURL.length === 0) {
 91         window.alert("Keine SKatDB-URL definiert");
 92         return;
 93       }
 94       url = dbURL + path;
 95       jQuery.ajax({
 96         url : url,
 97         type : 'GET',
 98         crossDomain : true,
 99         dataType : 'json',
100         error : function(xhr, status) {
101           window.alert("Fehler: " + xhr.status + " (" + xhr.statusText + ")");
102           if (window.confirm('URL "' + url + '" laden?')) {
103             window.location.href = url;
104           }
105         },
106         success : function(json) {
107           success.apply(that, [ json ]);
108         }
109       });
110     },
111     /**
112      * @description transmits all local stored games to SkatDB
113      */
114     uploadGames : function() {
115       var dbURL = this.load("dbURL"), games, that = this;
116       if (dbURL === null || dbURL.length === 0) {
117         window.alert("Keine SKatDB-URL definiert");
118         return;
119       }
120       games = this.getExport();
121       if (games.length === 0) {
122         window.alert("Keine lokalen Spiele zum Übertragen vorhanden.");
123         return;
124       }
125       jQuery.ajax({
126         url : dbURL + api.games,
127         type : 'POST',
128         crossDomain : true,
129         data : {
130           games : JSON.stringify(games)
131         },
132         dataType : 'json',
133         error : function(xhr, status) {
134           window.alert("Fehler: " + xhr.status + " (" + xhr.statusText + "):\n" + xhr.responseText);
135           // console.log(xhr);
136         },
137         success : function(json) {
138           var uploaded = games.length, failedGames;
139           if (json.handList) {
140             failedGames = json.handList;
141             that._skat.store("games", failedGames);
142             uploaded -= failedGames.length;
143           } else {
144             that._skat.settings.removeGames();
145           }
146           window.alert(uploaded + " von " + games.length + " Spielen hochgeladen:\n" + json.msg);
147         }
148       });
149     }
150   });
151   SkatSync.api = api;
152   return SkatSync;
153 });
