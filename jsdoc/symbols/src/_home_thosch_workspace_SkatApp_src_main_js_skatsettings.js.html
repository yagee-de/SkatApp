 18 define("SkatSettings",
 19     [ "jquery", "jqm-init", "jquery.mobile", "SkatStorage" ],
 20     function(jQuery, jqmInit, jqm, SkatStorage) {
 21       "use strict";
 22       var SkatSettings = SkatStorage
 23           .extend(
 24           /** @lends SkatSettings.prototype */
 25           {
 26             /** 
 27              * @constructs
 28              * @extends SkatStorage 
 29              */
 30             init : function() {
 31             },
 32             /**
 33              * @description initialize settings page with current stored values (call this before displaying #settingsPage)
 34              */
 35             initSettings : function() {
 36               // dbURL
 37               var skatURL = this.load("dbURL"), playerList = jQuery("#playerList"), players = (this.load("players") || []), groupList = jQuery("#groupList"), groups = (this
 38                   .load("groups") || []);
 39               jQuery("#dbURL").val(skatURL);
 40               // players
 41               jQuery("#playerList .player").remove();
 42               jQuery.each(players, function(i, player) {
 43                 jQuery('<li class="player"/>').text(player).appendTo(playerList);
 44               });
 45               playerList.listview("refresh");
 46               // groups
 47               jQuery("#groupList .group").remove();
 48               jQuery.each(groups, function(i, group) {
 49                 jQuery('<li class="group"/>').text(group).appendTo(groupList);
 50               });
 51               groupList.listview("refresh");
 52             },
 53             /**
 54              * @description adds a player to local storage
 55              * @param {String} playerName name of player
 56              */
 57             addPlayer : function(playerName) {
 58               var players = (this.load("players") || []), i;
 59               for (i = 0; i < players.lenght; i++) {
 60                 if (players[i] === playerName) {
 61                   return;
 62                 }
 63               }
 64               players.push(playerName);
 65               players.sort();
 66               this.store("players", players);
 67             },
 68             /**
 69              * @description adds a group to local storage
 70              * @param {String} groupName name of group
 71              */
 72             addGroup : function(groupName) {
 73               var groups = (this.load("groups") || []), i;
 74               for (i = 0; i < groups.lenght; i++) {
 75                 if (groups[i] === groupName) {
 76                   return;
 77                 }
 78               }
 79               groups.push(groupName);
 80               this.store("groups", groups);
 81             },
 82             /**
 83              * @description sets the URL to SkatDB in local storage
 84              * @param {String} url URL to SkatDB
 85              * @see <a href="https://github.com/Mewel/SkatDB">SkatDB project page</a>
 86              */
 87             setDbURL : function(url) {
 88               this.store("dbURL", url);
 89             }
 90           });
 91       return SkatSettings;
 92     });
