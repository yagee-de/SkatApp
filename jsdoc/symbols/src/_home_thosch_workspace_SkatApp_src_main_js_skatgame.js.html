 18 define("SkatGame", [ "Class" ], function(Class) {
 19   "use strict";
 20   var month, day, bids, SkatGame;
 21   /**
 22    * @memberOf SkatGame
 23    * @constant
 24    * @description maps javascript month to names
 25    */
 26   month = [ "Januar", "Februar", "MÃ¤rz", "April", "Mai", "Juni", "Juli", "August", "September", "Oktober", "November", "Dezember" ];
 27   /**
 28    * @memberOf SkatGame
 29    * @constant
 30    * @description maps javascript days to names
 31    */
 32   day = [ "Sonntag", "Montag", "Dienstag", "Mittwoch", "Donnerstag", "Freitag", "Sonnabend" ];
 33   /**
 34    * @memberOf SkatGame
 35    * @constant
 36    * @description valid bid values
 37    */
 38   bids = [ 18, 20, 22, 23, 24, 27, 30, 33, 35, 36, 40, 44, 45, 46, 48, 50, 54, 55, 59, 60, 63, 66, 70, 72, 77, 80, 81, 84, 88, 90, 96, 99,
 39       100, 108, 110, 117, 120, 121, 126, 130, 132, 135, 140, 143, 144, 150, 153, 154, 156, 160, 162, 165, 168, 170, 176, 180, 187, 192,
 40       198, 204, 216, 240, 264 ];
 41   
 42   SkatGame = Class.extend(
 43   /** @lends SkatGame.prototype */
 44   {
 45     /** @constructs */
 46     init : function(json) {
 47       var that = this, key;
 48       if (typeof json === "string") {
 49         JSON.parse(json, function(key, value) {
 50           that.reviver(key, value);
 51         });
 52       } else if (typeof json === "object") {
 53         for (key in json) {
 54           this.reviver(key, json[key]);
 55         }
 56       } else {
 57         this.created = new Date();
 58         this.modified = new Date();
 59       }
 60     },
 61     /**
 62      * @field
 63      * @type String
 64      */
 65     group : null,
 66     /**
 67      * @field
 68      * @type String
 69      */
 70     player : null,
 71     /**
 72      * @field
 73      * @type int
 74      */
 75     bid : 0,
 76     /**
 77      * @field
 78      * @type int
 79      */
 80     gameLevel : 0,
 81     /**
 82      * @field
 83      * @type Date
 84      */
 85     createDate : 0,
 86     /**
 87      * @field
 88      * @type Date
 89      */
 90     modifydDate : 0,
 91     /**
 92      * @field
 93      * @type int
 94      */
 95     points : 0,
 96     /**
 97      * @field
 98      * @type int
 99      */
100     jacks : 0,
101     /**
102      * @field
103      * @type int
104      */
105     gameType : 0,
106     /**
107      * @field
108      * @type boolean
109      */
110     hand : false,
111     /**
112      * @field
113      * @type boolean
114      */
115     won : true,
116     /**
117      * @field
118      * @type int
119      */
120     announcement : 1,
121     /**
122      * @description returns JSON representation of this game
123      * @Type Object
124      */
125     toJSON : function() {
126       var json = {
127         "group" : this.group,
128         "player" : this.player,
129         "bid" : this.bid,
130         "gameLevel" : this.gameLevel,
131         "createDate" : this.createDate.getTime(),
132         "modifyDate" : this.modifyDate.getTime()
133       };
134       if (this.isRamsch()) {
135         json.points = (this.gameLevel === 2) ? 120 : this.points;
136         json.won = (this.gameLevel === 2);
137       } else {
138         json.jacks = this.jacks;
139         json.gameType = this.gameType;
140         json.hand = this.hand;
141         json.announcement = this.announcement;
142         json.won = this.won;
143       }
144       return json;
145     },
146     /**
147      * @description used by JSON.parse to fill this instance
148      */
149     reviver : function(key, value) {
150       switch (key) {
151       case "group":
152       case "player":
153         if (typeof value === 'string') {
154           this[key] = value;
155         } else {
156           throw key + " is not a string";
157         }
158         break;
159       case "bid":
160         if (typeof value === 'number') {
161           var valid = (value === 0), i;
162           if (!valid) {
163             for (i = 0; i < bids.length; i++) {
164               if (bids[i] === value) {
165                 valid = true;
166                 break;
167               }
168             }
169           }
170           if (!valid) {
171             throw "bid has not a valid value";
172           }
173           this.bid = value;
174         } else {
175           throw key + " is not a number";
176         }
177         break;
178       case "gameLevel":
179       case "jacks":
180       case "gameType":
181       case "announcement":
182       case "points":
183         if (typeof value === 'number') {
184           this[key] = value;
185         } else {
186           throw key + " is not a number";
187         }
188         break;
189       case "won":
190       case "hand":
191         if (typeof value === 'boolean') {
192           this[key] = value;
193         } else {
194           throw key + " is not a boolean value";
195         }
196         break;
197       case "createDate":
198       case "modifyDate":
199         if (typeof value === 'number' && value > new Date(2000, 0, 1, 0, 0, 0, 0).getTime()) {
200           this[key] = new Date(value);
201         } else {
202           throw key + " is not a valid date value";
203         }
204         break;
205 
206       default:
207         window.alert("Ignored unknown key: " + key + ", value: " + value);
208         break;
209       }
210     },
211     /**
212      * @description returns true if this game is a Ramsch game (bid is 0)
213      * @type boolean
214      */
215     isRamsch : function() {
216       return this.bid === 0;
217     },
218     /**
219      * @returns {String} name of this game type
220      */
221     getGameTypeName : function() {
222       switch (this.gameType) {
223       case 9:
224         return "Schellen";
225       case 10:
226         return "Rot";
227       case 11:
228         return "GrÃ¼n";
229       case 12:
230         return "Eichel";
231       case 23:
232         return "Null";
233       case 24:
234         return "Grand";
235       case 35:
236         return "Null Hand";
237       case 46:
238         return "Null Ouvert";
239       case 59:
240         return "Null Ouvert Hand";
241       default:
242         return "unbekannt";
243       }
244     },
245     /**
246      * @description translates {@link SkatGame#createDate} into human readable time
247      * @see SkatGame#getDateString
248      * @type String
249      */
250     getTimeString : function() {
251       var date = new Date(this.createDate);
252       return (date.getHours() < 10 ? "0" : "") + date.getHours() + ":" + (date.getMinutes() < 10 ? "0" : "") + date.getMinutes();
253     },
254     /**
255      * @description translates {@link SkatGame#createDate} into human readable date
256      * @see SkatGame#getTimeString
257      * @type String
258      */
259     getDateString : function() {
260       var date = new Date(this.createDate);
261       return day[date.getDay()] + ", " + date.getDate() + ". " + month[date.getMonth()] + " " + date.getFullYear();
262     },
263     /**
264      * @description calculates the points of this game
265      * @type int
266      */
267     getPoints : function() {
268       var gamePoints, gameType = this.gameType;
269       if (this.isRamsch()) {
270         if (this.gameLevel === 2) {
271           // Durchmarsch
272           return 240;
273         }
274         return this.points * this.gameLevel;
275       } else if (gameType === 23 || gameType === 35 || gameType === 46 || gameType === 59) {
276         gamePoints = gameType;
277       } else {
278         gamePoints = gameType * (this.jacks + this.gameLevel);
279       }
280       gamePoints *= this.announcement;
281       if (!this.won) {
282         gamePoints *= -2;
283       }
284       return gamePoints;
285     }
286   });
287   SkatGame.bids = bids;
288   SkatGame.month = month;
289   SkatGame.day = day;
290   return SkatGame;
291 });
